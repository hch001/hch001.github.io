create table Actor

  • 有多个参数的存储过程
-- 搜寻某个演员的所有电影评分，使用多个参数
-- 在命令行中可以先执行 delimiter //
create procedure ratingProc(IN name varchar(50), OUT rating float, OUT nums int)
begin
    select avg(film.rating),count(*) 
    from actor,perform,film
    where (actor.name = name and
        actor.actorID = perform.actorID 
        and perform.filmID = film.filmID)
    into rating ,nums;
end //  -- // 在非命令行中可以不需要

-- 使用
call ratingProc('张涵予', @rating_avg, @nums) //
-- 输出
select @rating_avg, @nums //
  • 复杂智能函数
-- 创建一个复杂一点的函数，多了一个可选项，是否忽略2010年前的数据

create procedure ratingProc(
    IN  name        varchar(50),
    IN  ifIgnore      boolean,
    OUT rating      float,
    OUT nums        int
) COMMENT 'calculating the average rating of all movies the actor played, optionally ignoring those released before 2010. '
begin
    -- Declare local variables for rating, nums
    declare rating_tmp float;
    declare nums_tmp int;

    -- ignore ?
    IF ifIgnore THEN
        -- 和上面一样
        select avg(film.rating),count(*) 
        from actor,perform,film
        where (
            film.releaseDate >2010-01-01 
            and actor.name = name 
            and actor.actorID = perform.actorID 
            and perform.filmID = film.filmID 
        )
        into rating_tmp, nums_tmp;
    
    ELSE 
        select avg(film.rating),count(*) 
        from actor,perform,film
        where (
            actor.name = name 
            and actor.actorID = perform.actorID 
            and perform.filmID = film.filmID 
        )
        into rating_tmp, nums_tmp;
    END IF;

    select rating_tmp,nums_tmp into rating,nums;
end;

-- 调用
call ratingProc('张涵予',false,@rating,@num); -- 使用 0 也能代表 false
select @rating,@num;

游标

  • mysql 的游标仅用于存储过程和函数
create procedure procedure1()
begin
    -- 声明变量
    declare done boolean default 0;
    declare id varchar(10);
    declare n varchar(50);
    declare b date;

    -- 声明一个游标
    declare cursor1 cursor
    for
    select actorID,name,birthday from actor where birthday > '1990-01-01' limit 100;

    -- 如果没有取到数据设置 done = 1
    declare continue handler for not found set done = 1;

    -- open the cursor
    open cursor1;
    
    -- 创建一个输出的表
    create table young 
    (
        ID varchar(10) not null,
        name varchar(50) ,
        birthday date not null,
        primary key(ID)
    );

    -- 循环插入表
    myloop:loop 
        fetch cursor1 into id,n,b;  

        if done THEN
            LEAVE myloop;
        end if;

        insert into young values(id,n,b);
    end loop;

    -- 关闭游标
    close cursor1;

end;

触发器

  • 只有表支持触发器，视图不支持
  • 每个表最多支持六个触发器，INSERT，UPDATE，DELETE的之前和之后
-- 每添加一个就输出提示
create trigger newComment after insert on Actor
for each row select `Commment added!`;
  • NEW 和 OLD 关键字
    • NEW 指 insert 和 update 中的临时数据，使用 before 可以更改NEW中的信息，after只读
    • OLD 指 delete 和 update 执行之前或者之后（即使执行了delete和UPDATE 仍然能访问OLD），将要被处理的旧数据，只读，不能被更新。
    -- insert NEW
-- 在输入之前进行检查
create trigger parseAge before INSERT on test
for EACH Row 
BEGIN
    if NEW.age <0 THEN 
        set NEW.age = 0;
    elseif NEW.age > 200 THEN
        set NEW.age = 0;
    END IF;
end;


-- delete
-- 实现回收站的功能
create trigger deleteRows after INSERT on test
for each row
begin
    insert into trash values(OLD.id,OLD.age,OLD.name);
end;

-- update
create trigger parseRow before UPDATE on test
for each row
begin
    insert into trash values(OLD.id,OLD.age,OLD.name);
    set NEW.name = Upper(NEW.name)
end;

模式 描述
^ 匹配输入字符串的开始位置。如果设置了 RegExp 对象的 Multiline 属性，^ 也匹配 ‘\n’ 或 ‘\r’ 之后的位置。
$ 匹配输入字符串的结束位置。如果设置了RegExp 对象的 Multiline 属性，$ 也匹配 ‘\n’ 或 ‘\r’ 之前的位置。
. 匹配除 “\n” 之外的任何单个字符。要匹配包括 ‘\n’ 在内的任何字符，请使用像 ‘[.\n]’ 的模式。
[…] 字符集合。匹配所包含的任意一个字符。例如， ‘[abc]’ 可以匹配 “plain” 中的 ‘a’。
[^…] 负值字符集合。匹配未包含的任意字符。例如， ‘[^abc]’ 可以匹配 “plain” 中的’p'。
p1|p2|p3 匹配 p1 或 p2 或 p3。例如，‘z
* 匹配前面的子表达式零次或多次。例如，zo* 能匹配 “z” 以及 “zoo”。* 等价于{0,}。
+ 匹配前面的子表达式一次或多次。例如，‘zo+’ 能匹配 “zo” 以及 “zoo”，但不能匹配 “z”。+ 等价于 {1,}。
{n} n 是一个非负整数。匹配确定的 n 次。例如，‘o{2}’ 不能匹配 “Bob” 中的 ‘o’，但是能匹配 “food” 中的两个 o。
{n,m} m 和 n 均为非负整数，其中n <= m。最少匹配 n 次且最多匹配 m 次。

使用全文本搜索

  • mysql 支持两种数据库引擎 MyISAM 和 Innodb，只有前者支持全文本搜索
create table notes
(
    note_id     int     not null auto_increment, -- 从1开始
    note_text   text    null,
    primary key(note_id),
    fulltext(note_text)   -- 引擎会创建索引
) engine=MyISAM;
  • 应该在导入数据完毕后再定义为 fulltext，提高效率
  • 使用 Match() 和 Against() 进行搜索
select note_text
from notes
where Match(note_text) Against('rabbit');
  • 与使用正则表达式的区别在于这个返回的结果是进行排序过的，词出现的频率，出现的位置等
