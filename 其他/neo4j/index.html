SQL--Neo4j(基础用法)

May 28, 2020 · 3 min read

配置

windows配置neo4j

在windows环境中配置Neo4j

语法

在windows系统中执行，neo4j.bat console，就可以在网页中进行操作

创建/删除节点

create (n:Label1:Label2:Label3 {attr1:value1, attr2:value2}) 
-- n 代表别名如果不需要引用可省略
-- label 也可以省略

-- 例1
create (:Person {name:'Lee', age:20}),(:Person {name:'David', age:20})
-- 使用逗号创建多个节点

-- 删除节点，将会删除所有匹配的节点，如果存在关系则无法删除需要用到(detach)级联删除
MATCH (p:Person{name:'Lee'})
detach delete p

创建/删除关系

MATCH (a:Person),(b:Person)
where a.name = 'Lee' and b.name = 'David'
create (a)-[r:IS_FRIEND_OF {attr1:value1,attr2:value2}]->(b)
-- 创建的关系是单向的

MATCH (a:Person {name:'Lee'}),(b:Person {name:'David'})
create (a)-[r:IS_FRIEND_OF {attr1:value1,attr2:value2}]->(b)

-- 删除关系
MATCH (a:Person {name:'Lee'})-[r:IS_FRIEND_OF]->(b:Person {name:'David'})
delete r

添加/删除标签

MATCH(n {name:'Lee'}) 
where ...  
set n:Person :Worker -- 为节点添加两个标签

MATCH(n {name:'Lee'}) 
where ...  
remove n:Person -- 删除节点的标签

新增/修改/删除节点属性

MATCH(n:Person{name:'Lee'}) 
where ...  
set n+={age:30} -- 增加属性age

MATCH(n:Person{name:'Lee'}) 
where ...  
set n.age=15 -- 修改节点属性age

MATCH(n:Person{name:'Lee'}) 
where ...  
set n={age:20} -- 替换节点，覆盖掉原有的

MATCH(n:Person{name:'Lee'}) 
where ...  
remove n.age -- 删除一个属性

查看/创建/删除索引

create index on :Person(name) -- 创建索引

drop index on :Person(name) -- 删除索引

:schema -- 输出索引

查询

MATCH (a:Person)
where a.age<3
return distinct(a) as babies 
order by age asc/desc  -- asc是默认的

MATCH (a:Person)-[:IS_FRIEND_OF]-(b:Person) -- 这里无箭头说明a是b的朋友或者b是a的朋友
where a.name='Lee'
return a

-- 使用匿名以及多个匹配规则
MATCH (a:Person)-[:IS_FRIEND_OF]-()-[:HAS_SEEN]->(movie) ,
(a)-[:HAS_SEEN]->(movie)
return movie

条件

  • NOT
MATCH ...
where NOT ...
  • AND(逻辑与) |(逻辑或)
-- 使用了逻辑或
MATCH (a:Person{name:"lily"})-[:IS_FRIEND_OF | :SAME_AGE]->(b:Person)
return b

-- 使用逻辑与
MATCH (a:Person)
where a.age > 18 and a.name="lily"
return a
  • UNION
MATCH (a:Person)  where  n.age<3  return a.name as name
Union all -- (不去重)
MATCH (b:Person) where  n.age>80  return  b.name as name
-- 这里返回值的命名要相同
  • IN
MATCH (a:Person) 
where a.age IN [40,50,60,70,80,90,100]
return a.name
  • Limit 与 skip
-- Limit 代表一页的数据量
-- skip  代表跳过的数据
MATCH ...
return ...
skip 20
limit 10
-- 跳过了前20个匹配的数据，返回第21-20个
  • Starts with / End withs / Contains
MATCH (a:Person) 
where a.name starts/ends with 'L' -- 以L开头或者结尾的名字
-- a.name contains 'L' 名字中包含了L
return a.name as name
  • With
-- With 将一个语句的结果作为另一个语句的输入
MATCH (a:Person)-[:IS_FRIEND_OF]->(b:Person)-[:HAS_SEEN]->(c)
where id(a) = 10
with c, b as friends  -- 将上一条语句的c和b作为输入
where friends.age > 18
return c.title
  • 正则表达式
    • =~ 开启
    • (?i) 忽略大小写
    • (?m) 多行
MATCH (p:Person)
where p.name =~ '(?i)L.*' -- 忽略大小写，匹配以L开头的名字
return p.name

深度匹配

MATCH (a:Person)-[:IS_FRIEND_OF*1..2]->(b) return b -- 深度1到2

进阶用法

聚合

MATCH (a:Person)-[:IS_FRIEND_OF]-()
return a,count(*)
order by count(*) desc -- 输出朋友的数目，由大到小

MATCH ()-[r]-()
return type(r),count(*) -- 输出

列表和With

-- 使用[begin..end] 相等于[begin:end]
WITH ['Anne', 'John', 'Bill', 'Diane', 'Eve'] AS names
RETURN names[1..3] AS result

-- + 拼接，适用于列表和字符串
WITH ['Anne', 'John', 'Bill', 'Diane', 'Eve'] AS names
RETURN names[1..3]+[1,2] AS result

-- IN
RETURN [13]+[1,2] AS result -- 输出  [13,1,2]

-- 嵌套列表
return [1] in [[1],3,5] as result -- 输出 true

版本检查

-- 在语句前加上使用CYPHER 1.8

省略路径

  • 写成(a)–>(b)
MATCH p=(a)-->(b)
return p

unwind

-- 遍历列表输出 每个元素输出一行
unwind [1,2,3] as x
return x

-- 嵌套
unwind [[1,2],3,4] as x
unwind x as y
return y

foreach列表更新

match p=(a)-->(b)
foreach (n in nodes(p) | set n.age=100)

使用jdbc操作

  • 下载依赖
        <dependency>
            <groupId>org.neo4j</groupId>
            <artifactId>neo4j-jdbc-driver</artifactId>
            <version>3.1.0</version>
        </dependency>
  • 启动
    public static  void main(String[] args)
    {
        try {
            // Class.forName("org.neo4j.jdbc.Driver");
            Connection conn = DriverManager.getConnection("jdbc:neo4j:http://localhost:7474/",userName,password);
            // 这里用的 http
            conn.setAutoCommit(false);
            PreparedStatement preparedStatement = conn.prepareStatement("create (:Person{name:'lily',age:20})");
            preparedStatement.execute();
            conn.commit();

        }
        catch(Exception e)
        {
            e.printStackTrace();
        }
    }
